#include <thread>

#include <chrono>

#include <algorithm>
#include <cstring>
#include <fstream>
#include <sstream>

#ifdef _WIN32
    #include <windows.h>
    #include <psapi.h>
    #pragma comment(lib, "psapi.lib")
#elif __APPLE__
    #include <mach/mach.h>
    #include <mach/mach_time.h>
    #include <sys/sysctl.h>
#elif __linux__
    #include <sys/sysinfo.h>
    #include <unistd.h>
#endif

class SystemOptimizer {
private:
    struct OptimizationConfig {
        float touchSensitivity = 1.2f;
        float touchResponseBoost = 1.5f;
        bool enableTouchPrediction = true;
        float touchLatencyReduction = 0.3f;
        
        int targetFPS = 60;
        bool enableCPUBoost = true;
        bool enableGPUBoost = true;
        float cpuGovernorAggressive = 0.8f;
        float gpuClockBoost = 1.15f;
        
        bool autoClearCache = true;
        size_t cacheClearThresholdMB = 100;
        bool memoryCompression = true;
        
        bool reduceBackgroundProcesses = true;
        bool optimizePowerProfile = true;
        bool disableAnimations = false;
    } config;
    
    struct SystemState {
        int currentFPS = 0;
        float cpuUsage = 0.0f;
        float gpuUsage = 0.0f;
        size_t memoryUsageMB = 0;
        float touchLatencyMs = 0.0f;
        float frameTimeMs = 0.0f;
        int droppedFrames = 0;
    } state;
    
    std::atomic<bool> isOptimizing{false};
    std::atomic<bool> emergencyMode{false};
    std::thread optimizationThread;
    
    std::vector<float> fpsHistory;
    std::vector<float> latencyHistory;
    
    class TouchOptimizer {
    private:
        struct TouchPoint {
            float x, y;
            uint64_t timestamp;
            float pressure;
        };
        
        std::vector<TouchPoint> touchHistory;
        const size_t maxHistory = 10;
        
        struct KalmanFilter {
            float Q = 0.01f;
            float R = 0.1f;
            float x = 0.0f;
            float P = 1.0f;
            float K = 0.0f;
            
            float update(float measurement) {
                P = P + Q;
                K = P / (P + R);
                x = x + K * (measurement - x);
                P = (1 - K) * P;
                return x;
            }
        };
        
        KalmanFilter filterX, filterY;
        
    public:
        TouchOptimizer() = default;
        
        std::pair<float, float> processTouch(float x, float y, float pressure, uint64_t timestamp) {
            float filteredX = filterX.update(x);
            float filteredY = filterY.update(y);
            
            if (touchHistory.size() >= 2) {
                const auto& last = touchHistory.back();
                const auto& secondLast = touchHistory[touchHistory.size() - 2];
                
                float dt = static_cast<float>(timestamp - last.timestamp) / 1000000.0f;
                if (dt > 0) {
                    float velX = (last.x - secondLast.x) / dt;
                    float velY = (last.y - secondLast.y) / dt;
                    
                    filteredX += velX * 0.008f;
                    filteredY += velY * 0.008f;
                }
            }
            
            touchHistory.push_back({filteredX, filteredY, timestamp, pressure});
            if (touchHistory.size() > maxHistory) {
                touchHistory.erase(touchHistory.begin());
            }
            
            return {filteredX, filteredY};
        }
        
        std::pair<float, float> applySensitivity(float x, float y, float sensitivity) {
            float centerX = 0.5f;
            float centerY = 0.5f;
            
            float dx = x - centerX;
            float dy = y - centerY;
            float distance = std::sqrt(dx*dx + dy*dy);
            
            float curve = 1.0f + (distance * sensitivity * 0.5f);
            
            return {
                centerX + dx * curve,
                centerY + dy * curve
            };
        }
        
        void clearHistory() {
            touchHistory.clear();
            filterX = KalmanFilter();
            filterY = KalmanFilter();
        }
    };
    
    TouchOptimizer touchOptimizer;
    
    class MemoryManager {
    private:
        std::vector<void*> allocatedCache;
        size_t totalCacheSize = 0;
        const size_t maxCacheSizeMB = 50;
        
    public:
        MemoryManager() = default;
        
        void autoClearIfNeeded(size_t currentUsageMB) {
            if (currentUsageMB > maxCacheSizeMB) {
                clearUnusedCache();
            }
        }
        
        void clearUnusedCache() {
            #ifdef _WIN32
                SetProcessWorkingSetSize(GetCurrentProcess(), SIZE_MAX, SIZE_MAX);
            #elif __linux__
                std::ofstream("/proc/sys/vm/drop_caches") << "3";
            #endif
            
            allocatedCache.clear();
            totalCacheSize = 0;
        }
        
        void compressMemory() {
            std::vector<void*> newCache;
            newCache.reserve(allocatedCache.size());
            allocatedCache.swap(newCache);
        }
    };
    
    MemoryManager memoryManager;
    
    class PerformanceMonitor {
    private:
        uint64_t lastFrameTime = 0;
        int frameCount = 0;
        uint64_t lastFPSUpdate = 0;
        std::vector<float> frameTimes;
        
    public:
        PerformanceMonitor() {
            lastFrameTime = getCurrentTime();
            lastFPSUpdate = lastFrameTime;
        }
        
        uint64_t getCurrentTime() {
            #ifdef _WIN32
                LARGE_INTEGER time, freq;
                QueryPerformanceFrequency(&freq);
                QueryPerformanceCounter(&time);
                return static_cast<uint64_t>((time.QuadPart * 1000000) / freq.QuadPart);
            #else
                return std::chrono::duration_cast<std::chrono::microseconds>(
                    std::chrono::high_resolution_clock::now().time_since_epoch()
                ).count();
            #endif
        }
        
        void beginFrame() {
            lastFrameTime = getCurrentTime();
        }
        
        void endFrame() {
            uint64_t currentTime = getCurrentTime();
            float frameTime = static_cast<float>(currentTime - lastFrameTime) / 1000.0f;
            frameTimes.push_back(frameTime);
            
            frameCount++;
            
            if (currentTime - lastFPSUpdate >= 1000000) {
                if (!frameTimes.empty()) {
                    float avgFrameTime = 0;
                    for (float ft : frameTimes) avgFrameTime += ft;
                    avgFrameTime /= frameTimes.size();
                    
                    frameTimes.clear();
                }
                
                frameCount = 0;
                lastFPSUpdate = currentTime;
            }
        }
        
        float getAverageFPS() {
            if (frameTimes.empty()) return 0;
            
            float totalTime = 0;
            for (float ft : frameTimes) totalTime += ft;
            float avgFrameTime = totalTime / frameTimes.size();
            
            return avgFrameTime > 0 ? 1000.0f / avgFrameTime : 0;
        }
    };
    
    PerformanceMonitor perfMonitor;
    
    #ifdef _WIN32
    void optimizeWindows() {
        if (config.reduceBackgroundProcesses) {
            SetPriorityClass(GetCurrentProcess(), HIGH_PRIORITY_CLASS);
            
            SetThreadPriority(GetCurrentThread(), THREAD_PRIORITY_HIGHEST);
        }
        
        if (config.optimizePowerProfile) {
            system("powercfg -setactive 8c5e7fda-e8bf-4a96-9a85-a6e23a8c635c");
        }
    }
    #elif __linux__
    void optimizeLinux() {
        if (config.reduceBackgroundProcesses) {
            nice(-10);
            
            struct sched_param param;
            param.sched_priority = 50;
            sched_setscheduler(0, SCHED_FIFO, &param);
        }
        
        if (config.autoClearCache) {
            std::ofstream dropCaches("/proc/sys/vm/drop_caches");
            dropCaches << "3";
            dropCaches.close();
        }
    }
    #endif
    
    void optimizationLoop() {
        const std::chrono::milliseconds checkInterval(100);
        
        while (isOptimizing) {
            updateSystemState();
            
            applyDynamicOptimizations();
            
            if (emergencyMode) {
                applyEmergencyOptimizations();
            }
            
            if (config.autoClearCache && state.memoryUsageMB > config.cacheClearThresholdMB) {
                memoryManager.clearUnusedCache();
            }
            
            std::this_thread::sleep_for(checkInterval);
        }
    }
    
    void updateSystemState() {
        #ifdef _WIN32
            static ULARGE_INTEGER lastCPU, lastSysCPU, lastUserCPU;
            static int numProcessors = 0;
            
            FILETIME ftime, fsys, fuser;
            GetSystemTimeAsFileTime(&ftime);
            memcpy(&lastCPU, &ftime, sizeof(FILETIME));
            
            GetProcessTimes(GetCurrentProcess(), &ftime, &ftime, &fsys, &fuser);
            memcpy(&lastSysCPU, &fsys, sizeof(FILETIME));
            memcpy(&lastUserCPU, &fuser, sizeof(FILETIME));
            
            ULARGE_INTEGER now, sys, user;
            GetSystemTimeAsFileTime(&ftime);
            memcpy(&now, &ftime, sizeof(FILETIME));
            
            GetProcessTimes(GetCurrentProcess(), &ftime, &ftime, &fsys, &fuser);
            memcpy(&sys, &fsys, sizeof(FILETIME));
            memcpy(&user, &fuser, sizeof(FILETIME));
            
            float percent = (sys.QuadPart - lastSysCPU.QuadPart) + 
                            (user.QuadPart - lastUserCPU.QuadPart);
            percent /= (now.QuadPart - lastCPU.QuadPart);
            percent /= numProcessors;
            
            state.cpuUsage = percent * 100.0f;
        #endif
        
        state.currentFPS = static_cast<int>(perfMonitor.getAverageFPS());
        
        if (state.currentFPS < config.targetFPS * 0.8f) {
            state.droppedFrames++;
        }
        
        if (state.droppedFrames > 10 || state.cpuUsage > 90.0f) {
            emergencyMode = true;
        }
    }
    
    void applyDynamicOptimizations() {
        if (config.enableCPUBoost) {
            float loadFactor = state.cpuUsage / 100.0f;
            float boost = config.cpuGovernorAggressive * (1.0f + loadFactor * 0.5f);
            applyCPUBoost(boost);
        }
        
        if (state.currentFPS < config.targetFPS) {
            reduceGraphicsQuality();
            
            increaseProcessPriority();
        }
        
        if (config.memoryCompression && state.memoryUsageMB > 500) {
            memoryManager.compressMemory();
        }
    }
    
    void applyEmergencyOptimizations() {
        std::cout << "EMERGENCY OPTIMIZATION MODE ACTIVATED!" << std::endl;
        
        setMaxCPUPriority();
        
        config.disableAnimations = true;
        
        memoryManager.clearUnusedCache();
        
        setMinGraphicsSettings();
        
        killBackgroundProcesses();
        
        static auto emergencyStart = std::chrono::steady_clock::now();
        auto now = std::chrono::steady_clock::now();
        if (std::chrono::duration_cast<std::chrono::seconds>(now - emergencyStart).count() > 5) {
            emergencyMode = false;
        }
    }
    
    void applyCPUBoost(float boost) {
        #ifdef _WIN32
            DWORD_PTR affinityMask = 0xFF;
            SetProcessAffinityMask(GetCurrentProcess(), affinityMask);
        #endif
    }
    
    void reduceGraphicsQuality() {
        std::cout << "Reducing graphics quality..." << std::endl;
    }
    
    void increaseProcessPriority() {
        #ifdef _WIN32
            SetPriorityClass(GetCurrentProcess(), REALTIME_PRIORITY_CLASS);
        #endif
    }
    
    void setMaxCPUPriority() {
        #ifdef _WIN32
            SetPriorityClass(GetCurrentProcess(), REALTIME_PRIORITY_CLASS);
            SetThreadPriority(GetCurrentThread(), THREAD_PRIORITY_TIME_CRITICAL);
        #endif
    }
    
    void setMinGraphicsSettings() {
        std::cout << "Setting graphics to minimum..." << std::endl;
    }
    
    void killBackgroundProcesses() {
        std::cout << "Killing non-essential processes..." << std::endl;
    }
    
public:
    SystemOptimizer() {
        loadDefaultConfig();
    }
    
    ~SystemOptimizer() {
        stopOptimization();
    }
    
    void startOptimization() {
        if (isOptimizing) return;
        
        isOptimizing = true;
        optimizationThread = std::thread(&SystemOptimizer::optimizationLoop, this);
        
        std::cout << "System optimization started!" << std::endl;
    }
    
    void stopOptimization() {
        isOptimizing = false;
        if (optimizationThread.joinable()) {
            optimizationThread.join();
        }
        
        std::cout << "System optimization stopped!" << std::endl;
    }
    
    std::pair<float, float> processTouchInput(float x, float y, float pressure = 1.0f) {
        uint64_t timestamp = perfMonitor.getCurrentTime();
        
        auto sensitized = touchOptimizer.applySensitivity(x, y, config.touchSensitivity);
        
        auto processed = touchOptimizer.processTouch(
            sensitized.first, sensitized.second, 
            pressure, timestamp
        );
        
        float boost = config.touchResponseBoost;
        return {
            processed.first * boost,
            processed.second * boost
        };
    }
    
    void beginFrame() {
        perfMonitor.beginFrame();
    }
    
    void endFrame() {
        perfMonitor.endFrame();
        
        float currentFPS = perfMonitor.getAverageFPS();
        if (currentFPS < config.targetFPS * 0.9f) {
            reduceGraphicsQuality();
        }
    }
    
    void clearCacheNow() {
        memoryManager.clearUnusedCache();
        std::cout << "Cache cleared!" << std::endl;
    }
    
    void forceGarbageCollection() {
        #ifdef __ANDROID__
            system("am send-memory-pressure");
        #endif
    }
    
    void setTouchSensitivity(float sensitivity) {
        config.touchSensitivity = std::clamp(sensitivity, 0.5f, 3.0f);
    }
    
    void setTargetFPS(int fps) {
        config.targetFPS = std::clamp(fps, 30, 144);
    }
    
    void enableCPUBoosting(bool enable) {
        config.enableCPUBoost = enable;
    }
    
    void enableAutoCacheClear(bool enable) {
        config.autoClearCache = enable;
    }
    
    void printSystemStatus() {
        std::cout << "\n=== SYSTEM STATUS ===" << std::endl;
        std::cout << "FPS: " << state.currentFPS << "/" << config.targetFPS << std::endl;
        std::cout << "CPU Usage: " << state.cpuUsage << "%" << std::endl;
        std::cout << "Memory Usage: " << state.memoryUsageMB << " MB" << std::endl;
        std::cout << "Dropped Frames: " << state.droppedFrames << std::endl;
        std::cout << "Emergency Mode: " << (emergencyMode ? "ACTIVE" : "INACTIVE") << std::endl;
        std::cout << "=====================\n" << std::endl;
    }
    
    void applyGamingPreset() {
        std::cout << "Applying Gaming Preset..." << std::endl;
        
        config.touchSensitivity = 1.5f;
        config.touchResponseBoost = 1.8f;
        config.targetFPS = 60;
        config.enableCPUBoost = true;
        config.enableGPUBoost = true;
        config.autoClearCache = true;
        config.reduceBackgroundProcesses = true;
        config.disableAnimations = true;
        
        startOptimization();
    }
    
    void applyBatterySaverPreset() {
        std::cout << "Applying Battery Saver Preset..." << std::endl;
        
        config.touchSensitivity = 1.0f;
        config.targetFPS = 30;
        config.enableCPUBoost = false;
        config.enableGPUBoost = false;
        config.optimizePowerProfile = true;
        
        startOptimization();
    }
    
private:
    void loadDefaultConfig() {
        #ifdef __ANDROID__
            config.targetFPS = 90;
        #elif __APPLE__
            config.targetFPS = 120;
        #else
            config.targetFPS = 60;
        #endif
    }
};

int main() {
    SystemOptimizer optimizer;
    
    optimizer.applyGamingPreset();
    
    for (int i = 0; i < 1000; ++i) {
        optimizer.beginFrame();
        
        auto touchPos = optimizer.processTouchInput(0.5f, 0.5f);
        
        optimizer.endFrame();
        
        if (i % 100 == 0) {
            optimizer.printSystemStatus();
        }
        
        std::this_thread::sleep_for(std::chrono::milliseconds(16));
    }
    
    optimizer.clearCacheNow();
    
    optimizer.stopOptimization();
    
    return 0;
}
