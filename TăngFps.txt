using UnityEngine;

public class OptimizeFPS : MonoBehaviour
{
    void Start()
    {
        // Giảm tần suất cập nhật vật lý
        Time.fixedDeltaTime = 0.02f;

        // Giảm chất lượng đồ họa
        QualitySettings.SetQualityLevel(2); // 0 = Thấp, 5 = Cao

        // Tắt khử răng cưa
        QualitySettings.antiAliasing = 0;

        // Giới hạn FPS
        Application.targetFrameRate = 60;

        // Giảm khoảng cách render của camera
        Camera.main.farClipPlane = 100f;

        Debug.Log("FPS Optimization Applied!");
    }
}
